Как работает генератор?

Генератор написан на языке JavaScript и работает в браузере на стороне клиента. Функция генерации значения:


//...other code...

function getRandomNumber(min, max, includeMin, includeMax){
 let minCorrection = 0;
 let maxCorrection = 0;

 if(!includeMin)
  minCorrection = 1;
 if(!includeMax)
  maxCorrection = 1;

 let result = Math.floor(min + minCorrection + Math.random() * (max - min - maxCorrection));
 if(isNaN(result))
  result = "Not a number";
 return result;
}
